Goal: netsuite-cli saved-search run <id> --type <recordtype> executes a NetSuite saved search over SuiteTalk SOAP (search/searchMoreWithId), authenticated by TBA tokens minted through a browser flow (account soap-auth), with rows flattened to JSON.

Architecture: A self-contained src/soap/ module (hand-rolled envelopes, quick-xml parsing, static record-type table) plus src/auth/tba.rs (OAuth 1.0a signing + three-step browser flow reusing a generalized loopback listener extracted from authcode.rs). TBA credentials live in a new keyring entry per alias ({alias}#tba); nothing touches the existing REST client, providers, or exit-code contract. The consumer key/secret come from env vars, the keyring, or an interactive prompt — never from the binary.

Tech stack: Rust, clap v4 (derive), reqwest, quick-xml (new), hmac (new), rpassword (new), wiremock tests.

pub fn lookup(record_type: &str) -> Option<SearchType> {
    let wanted = record_type.to_ascii_lowercase();
    SEARCH_TYPES
        .iter()
        .find(|(cli_name, _, _)| *cli_name == wanted)
        .map(|(_, xsi_type, namespace)| SearchType { xsi_type, namespace })
}

pub fn known_type_names() -> String {
    SEARCH_TYPES
        .iter()
        .map(|(cli_name, _, _)| *cli_name)
        .collect::<Vec<_>>()
        .join(", ")
}

pub fn xml_escape(raw: &str) -> String {

}

pub async fn run(soap: &SoapClient, saved_search_id: &str, record_type: &str,
    limit: Option<u64>, fetch_all: bool) -> Result<Value, CliError>
{
    let search_type = search_types::lookup(record_type).ok_or_else(|| CliError::Usage(format!(
        "unknown record type '{record_type}'; valid types: {}",
        search_types::known_type_names())))?;

    while last_page_index < first_page.total_pages && last_page_index < MAX_PAGES {

        merged_rows.extend(next.rows);
        last_page_index = next.page_index;

        "totalRecords": first_page.total_records,

            .collect();

    alias: &str, port: u16, paste_mode: bool) -> Result<Value, CliError>
{

    })?;

pub fn resolve_consumer_pair(store: &dyn SecretStore, alias: &str)
    -> Result<(String, String), CliError>
{
    if let (Ok(env_key), Ok(env_secret)) = (
        std::env::var("NETSUITE_CLI_TBA_CONSUMER_KEY"),
        std::env::var("NETSUITE_CLI_TBA_CONSUMER_SECRET"),
    ) {
        return Ok((env_key, env_secret));
    }
    if let Some(stored) = store.get_tba(alias)? {
        return Ok((stored.consumer_key, stored.consumer_secret));
    }
    if !std::io::stdin().is_terminal() {
        return Err(CliError::Auth(format!(
            "no TBA consumer credentials for '{alias}': set NETSUITE_CLI_TBA_CONSUMER_KEY and \
             NETSUITE_CLI_TBA_CONSUMER_SECRET, or run `netsuite-cli account soap-auth {alias}` \
             in an interactive terminal")));
    }
    eprint!("Integration record consumer key (client id): ");
    let mut consumer_key = String::new();
    std::io::stdin().read_line(&mut consumer_key)
        .map_err(|read_error| CliError::Auth(format!("failed to read consumer key: {read_error}")))?;
    let consumer_secret = rpassword::prompt_password("Integration record consumer secret (hidden): ")
        .map_err(|read_error| CliError::Auth(format!("failed to read consumer secret: {read_error}")))?;
    Ok((consumer_key.trim().to_string(), consumer_secret.trim().to_string()))
}
